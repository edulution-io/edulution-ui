**useSseStore.spec.ts (FEST-08):**
Create a MockEventSource class at top of test file:

```typescript
class MockEventSource {
  static CONNECTING = 0;
  static OPEN = 1;
  static CLOSED = 2;
  readyState = MockEventSource.OPEN;
  url: string;
  private listeners = new Map<string, Set<Function>>();
  constructor(url: string) {
    this.url = url;
  }
  addEventListener(type: string, listener: Function, opts?: { signal?: AbortSignal }) {
    if (!this.listeners.has(type)) this.listeners.set(type, new Set());
    this.listeners.get(type)!.add(listener);
    opts?.signal?.addEventListener('abort', () => this.removeEventListener(type, listener));
  }
  removeEventListener(type: string, listener: Function) {
    this.listeners.get(type)?.delete(listener);
  }
  dispatchEvent(event: Event) {
    this.listeners.get(event.type)?.forEach((l) => l(event));
    return true;
  }
  close() {
    this.readyState = MockEventSource.CLOSED;
  }
}
```

Assign to globalThis.EventSource before tests. Also mock useUserStore to return an eduApiToken.
Tests:

- setEventSource: creates EventSource with token in URL
- setEventSource: does nothing when no eduApiToken
- ping event: updates lastPingTime and resets reconnectAttempts
- error event with CLOSED readyState: schedules reconnect
- reconnect: closes existing EventSource and creates new one
- reset: closes EventSource and resets state
